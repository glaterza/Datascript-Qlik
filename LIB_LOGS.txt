SUB LOGEXECINICIO

	TEMPLATE:
	LOAD
		NOW() 				AS TEMPLATE.INICIO,
		OSUSER()			AS TEMPLATE.USUARIO
	AUTOGENERATE 1;

	LEFT JOIN(TEMPLATE)
	LOAD
		'$(S_VERSION)' 		AS TEMPLATE.VERSION,
		'$(S_AUTOR)'		AS TEMPLATE.AUTOR,
		'$(S_FECHA)' 		AS TEMPLATE.FECHA,
		'$(S_DESCRIPCION)'	AS TEMPLATE.DESCRIPCION
	AUTOGENERATE 1;

END SUB

SUB LOGEXECFIN

	LEFT JOIN(TEMPLATE)
	LOAD
		NOW() AS TEMPLATE.FIN
	AUTOGENERATE 1;

	LEFT JOIN(TEMPLATE)
	LOAD
		TIME(TEMPLATE.FIN - TEMPLATE.INICIO, 'hh:mm:ss') AS TEMPLATE.DURACION
	RESIDENT TEMPLATE;
	

END SUB
